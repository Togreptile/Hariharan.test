
 class Main {
	boolean ispalindrome(String n)
	{
		boolean f = false;
		String s1 = "";
		for(int i=n.length()-1;i>=0;i--)
		{
			String s = n.substring(i,i+1);
			s1 = s1+s;
		}
		if(n.equals(s1))
		{
			f = true;
		}
		return f;
	}

	public static void main(String[] args) {
		Main t = new Main();
		System.out.println(t.ispalindrome("wow"));
	}	
	}
